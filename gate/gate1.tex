\title{GATE ECE 2023}
\author{Karyampudi Meghana Sai\\ EE23BTECH11031}
\maketitle
Consider a discrete-time signal with period $N=5$. Let the discrete-time Fourier series (DTFS) representation be $x[n]=\sum\limits_{k=0}^{4} a_k e^{\frac{jk2\pi n}{5}}$, where $a_0=1$, $a_1=3j$, $a_2=2j$, $a_3=-2j$, $a_4=-3j$. The value of the sum $\sum\limits_{n=0}^{4}x[n] \sin\brak{\frac{4\pi n}{5}}$ is\\


\solution
\begin{align}
\sum\limits_{n=0}^{4}x(n) \sin\brak{\frac{4\pi n}{5}}&=\sum\limits_{n=0}^{4}x[n]\sbrak{\frac{e^{\frac{j4\pi n}{5}}-e^{\frac{-j4\pi n}{5}}}{2j}}\label{eq:gate1eq1}\\
\sum\limits_{n=0}^{4}x(n) \sin\brak{\frac{4\pi n}{5}}&=\frac{1}{2j}\sbrak{\sum\limits_{n=0}^{4}x[n]e^{\frac{j2\pi (2)n}{5}}-\sum\limits_{n=0}^{4}x[n]e^{\frac{-j2\pi (2)n}{5}}}\label{eq:gate1eq2}
\end{align}
DFTS coefficient is given by,\\
\begin{align}
a_k=\frac{1}{N}\sum\limits_{n=0}^{N-1} x(n)e^{\frac{-j2\pi kn}{N}}\label{eq:gate1eq3}
\end{align}
Given that time period of x(n) is N=5 sec.\\
\begin{align}
a_k&=\frac{1}{5}\sum\limits_{n=0}^{4} x(n)e^{\frac{-j2\pi kn}{5}}\label{eq:gate1eq4}\\
\sum\limits_{n=0}^{4} x(n)e^{\frac{-j2\pi kn}{5}}&=5a_k\label{eq:gate1eq5}
\end{align}
Referencing from equation\eqref{eq:gate1eq5}, equation\eqref{eq:gate1eq2} can be written as:
\begin{align}
\sum\limits_{n=0}^{4}x(n) \sin\brak{\frac{4\pi n}{5}}&=\frac{1}{2j}\sbrak{5a_{-2}-5a_2}\label{eq:gate1eq6}
\end{align}
From the property of discrete Fourier series.\\
$a_k=a_{k+N}$,where N is the time period.\\
So, equation\eqref{eq:gate1eq6} becomes,\\
\begin{align}
\sum\limits_{n=0}^{4}x(n) \sin\brak{\frac{4\pi n}{5}}&=\frac{1}{2j}\sbrak{5a_3-5a_2}\label{eq:gate1eq7}\\
\sum\limits_{n=0}^{4}x(n) \sin\brak{\frac{4\pi n}{5}}&=-10\label{eq:gate1eq8}
\end{align}
